<script>
  const WORKFLOW_URL = 'YOUR_N8N_FORM_URL_HERE';

  function handleFile(input) {
    const file = input.files[0];
    if (!file) return;
    document.getElementById('uploadArea').classList.add('active');
    const info = document.getElementById('fileInfo');
    document.getElementById('fileInfoText').textContent = file.name + ' · ' + (file.size / 1024).toFixed(0) + ' KB';
    info.style.display = 'flex';
  }

  const area = document.getElementById('uploadArea');
  area.addEventListener('dragover', e => { e.preventDefault(); area.classList.add('active'); });
  area.addEventListener('dragleave', () => { if (!document.getElementById('resumeFile').files.length) area.classList.remove('active'); });
  area.addEventListener('drop', e => {
    e.preventDefault();
    const file = e.dataTransfer.files[0];
    if (file) { document.getElementById('resumeFile').files = e.dataTransfer.files; handleFile(document.getElementById('resumeFile')); }
  });

  async function handleSubmit(e) {
    e.preventDefault();
    const btn = document.getElementById('submitBtn');
    const spinner = document.getElementById('spinner');
    const btnText = document.getElementById('btnText');
    const btnArrow = document.getElementById('btnArrow');

    btn.style.pointerEvents = 'none';
    btn.style.opacity = '0.8';
    spinner.style.display = 'block';
    btnText.textContent = 'Launching AI Agent…';
    btnArrow.style.display = 'none';

    const formData = new FormData();
    formData.append('Full_Name', document.getElementById('fullName').value);
    formData.append('Email', document.getElementById('emailAddr').value);
    formData.append('Title', document.getElementById('jobTitle').value);
    formData.append('Country', document.getElementById('country').value);
    formData.append('Location', document.getElementById('location').value);
    formData.append('Your_Resume', document.getElementById('resumeFile').files[0]);

    try { await fetch(WORKFLOW_URL, { method: 'POST', body: formData }); } catch(err) {}

    const name = document.getElementById('fullName').value.split(' ')[0];
    const email = document.getElementById('emailAddr').value;
    document.getElementById('successMsg').textContent =
      `Hi ${name}! Your AI agent is now scanning listings and scoring matches. Results will be sent to ${email} in 1–3 minutes.`;

    document.getElementById('jobForm').style.display = 'none';
    document.getElementById('successState').style.display = 'flex';
  }
</script>
